
# nb のファイルを fzf で検索して nvim で開く関数
function nb_f() {
  local file=$(nb ls --all --paths | fzf --preview 'bat --color=always --style=numbers {}')
  [ -n "$file" ] && nvim "$file"
}

# nb のノートブックを fzf で切り替える
function nb_u() {
  local notebook=$(nb notebooks | fzf)
  [ -n "$notebook" ] && nb use "$notebook"
}

# 名前を nsearch (nb search の略) などに変更
function nb_g() {
  # nb notebooks --paths で全てのノートブックのパスを取得し、そこからファイルを検索
  local file=$(rg --files $(nb notebooks --paths) | fzf --preview 'bat --color=always --style=numbers {}')
  [ -n "$file" ] && nvim "$file"
}
